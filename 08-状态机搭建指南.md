# 基于 Spring Boot 3.5.9 与 Spring State Machine 4.0.0 的完整状态机实现方案

## 技术栈与版本说明

- **Spring Boot**: 3.5.9 (Spring Framework 6.x)
- **JDK**: 17+
- **Spring State Machine**: 4.0.0
- **数据库**: PostgreSQL 12+
- **持久化**: Spring Data JPA + 状态机 JPA 模块
- **构建工具**: Maven 3.6+

```java
package com.kidsky.statemachine.domain;

import lombok.Getter;

/**
 * 订单状态枚举
 * 定义了订单生命周期的所有可能状态
 */
@Getter
public enum OrderState {
    
    // 初始状态
    UNPAID("待支付", "订单已创建，等待用户支付"),
    
    // 处理中状态
    WAITING_PAYMENT_CONFIRM("等待支付确认", "用户已发起支付，等待第三方支付回调"),
    PAID("已支付", "支付成功，等待商家处理"),
    PACKAGED("已打包", "商品已打包完成，等待发货"),
    SHIPPED("已发货", "商品已发出，运输中"),
    DELIVERED("已送达", "商品已送达用户指定地点"),
    
    // 终态（不可再转换）
    CONFIRMED("已确认收货", "用户已确认收到商品"),
    COMPLETED("已完成", "订单完成所有流程"),
    CANCELLED("已取消", "订单已被取消"),
    REFUNDED("已退款", "订单已退款"),
    REFUND_FAILED("退款失败", "退款流程失败"),
    EXPIRED("已过期", "订单超过有效期");
    
    private final String name;
    private final String description;
    
    OrderState(String name, String description) {
        this.name = name;
        this.description = description;
    }
    
    /**
     * 检查是否为终态
     */
    public boolean isEndState() {
        return this == COMPLETED || 
               this == CANCELLED || 
               this == REFUNDED || 
               this == EXPIRED;
    }
    
    /**
     * 检查是否允许取消
     */
    public boolean canBeCancelled() {
        return this == UNPAID || 
               this == WAITING_PAYMENT_CONFIRM || 
               this == PAID;
    }
    
    @Override
    public String toString() {
        return this.name + "(" + this.name() + ")";
    }
}
```

### 第4步：JPA实体类

#### 1、Order.java  - 订单业务实体

```java
package com.kidsky.statemachine.entity;

import com.kidsky.statemachine.domain.OrderState;
import jakarta.persistence.*;
import lombok.Data;
import lombok.NoArgsConstructor;
import org.springframework.data.annotation.CreatedDate;
import org.springframework.data.annotation.LastModifiedDate;
import org.springframework.data.jpa.domain.support.AuditingEntityListener;

import java.math.BigDecimal;
import java.time.LocalDateTime;

/**
 * 订单业务实体
 * 存储订单的核心业务信息
 */
@Data
@NoArgsConstructor
@Entity
@Table(name = "orders", indexes = {
        @Index(name = "idx_order_number", columnList = "orderNumber", unique = true),
        @Index(name = "idx_user_id", columnList = "userId"),
        @Index(name = "idx_state", columnList = "state"),
        @Index(name = "idx_create_time", columnList = "createTime")
})
@EntityListeners(AuditingEntityListener.class)
public class Order {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "order_number", nullable = false, unique = true, length = 32)
    private String orderNumber;

    @Enumerated(EnumType.STRING)
    @Column(name = "state", nullable = false, length = 50)
    private OrderState state = OrderState.UNPAID;

    @Column(name = "amount", nullable = false, precision = 10, scale = 2)
    private BigDecimal amount;

    @Column(name = "user_id", nullable = false, length = 100)
    private String userId;

    @Column(name = "user_name", length = 100)
    private String userName;

    @Column(name = "product_id", length = 100)
    private String productId;

    @Column(name = "product_name", length = 200)
    private String productName;

    @Column(name = "quantity", nullable = false)
    private Integer quantity = 1;

    @Column(name = "recipient_name", length = 100)
    private String recipientName;

    @Column(name = "recipient_phone", length = 20)
    private String recipientPhone;

    @Column(name = "shipping_address", length = 500)
    private String shippingAddress;

    @Column(name = "remark", length = 500)
    private String remark;

    @CreatedDate
    @Column(name = "create_time", nullable = false, updatable = false)
    private LocalDateTime createTime;

    @LastModifiedDate
    @Column(name = "update_time", nullable = false)
    private LocalDateTime updateTime;

    @Column(name = "pay_time")
    private LocalDateTime payTime;

    @Column(name = "ship_time")
    private LocalDateTime shipTime;

    @Column(name = "complete_time")
    private LocalDateTime completeTime;

    @Version
    @Column(name = "version", nullable = false)
    private Integer version = 0;

    /**
     * 状态机ID - 用于关联状态机上下文
     * 格式：ORDER_{orderId}
     */
    @Column(name = "state_machine_id", length = 100)
    private String stateMachineId;

    /**
     * 计算状态机ID
     */
    @PrePersist
    public void prePersist() {
        if (this.orderNumber == null) {
            // 生成订单号：时间戳+随机数
            this.orderNumber = "ORD" + System.currentTimeMillis() + (int)(Math.random() * 1000);
        }

        if (this.stateMachineId == null) {
            // 对于新订单，暂时设置一个临时ID，保存后更新
            this.stateMachineId = "ORDER_TEMP_" + this.orderNumber;
        }
    }

    /**
     * 保存后更新状态机ID为实际ID
     */
    @PostPersist
    public void postPersist() {
        this.stateMachineId = "ORDER_" + this.id;
    }

    /**
     * 创建订单的工厂方法
     */
    public static Order createOrder(String userId, String userName, BigDecimal amount,
                                    String productId, String productName) {
        Order order = new Order();
        order.setUserId(userId);
        order.setUserName(userName);
        order.setAmount(amount);
        order.setProductId(productId);
        order.setProductName(productName);
        return order;
    }
}
```

```java
package com.kidsky.statemachine.repository;

import com.kidsky.statemachine.domain.OrderState;
import com.kidsky.statemachine.entity.Order;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;
import org.springframework.stereotype.Repository;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;

@Repository
public interface OrderRepository extends JpaRepository<Order, Long> ,
        JpaSpecificationExecutor<Order> {
    
    /**
     * 根据订单号查找订单
     */
    Optional<Order> findByOrderNumber(String orderNumber);
    
    /**
     * 根据用户ID查找订单（按创建时间倒序）
     */
    List<Order> findByUserIdOrderByCreateTimeDesc(String userId);
    
    /**
     * 根据状态查找订单
     */
    List<Order> findByState(OrderState state);
    
    /**
     * 查找指定状态且创建时间早于指定时间的订单
     * 用于定时任务处理超时订单
     */
    List<Order> findByStateAndCreateTimeBefore(OrderState state, LocalDateTime time);
    
    /**
     * 查找待支付超时的订单
     */
    @Query("SELECT o FROM Order o WHERE o.state = :state AND o.createTime < :timeoutTime ORDER BY o.createTime ASC")
    List<Order> findTimeoutOrders(@Param("state") OrderState state, 
                                 @Param("timeoutTime") LocalDateTime timeoutTime);
    
    /**
     * 根据状态机ID查找订单
     */
    Optional<Order> findByStateMachineId(String stateMachineId);
    
    /**
     * 批量更新订单状态
     */
    @Query("UPDATE Order o SET o.state = :newState, o.updateTime = CURRENT_TIMESTAMP WHERE o.id IN :ids")
    int batchUpdateState(@Param("ids") List<Long> ids, @Param("newState") OrderState newState);
    
    /**
     * 统计用户订单数量
     */
    long countByUserId(String userId);
}
```

#### 2、OrderStateMachineConfig.java  - 状态机核心配置

```java
package com.kidsky.statemachine.config;

import com.kidsky.statemachine.domain.OrderEvent;
import com.kidsky.statemachine.domain.OrderState;
import lombok.extern.slf4j.Slf4j;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.statemachine.StateMachine;
import org.springframework.statemachine.action.Action;
import org.springframework.statemachine.config.EnableStateMachineFactory;
import org.springframework.statemachine.config.StateMachineConfigurerAdapter;
import org.springframework.statemachine.config.builders.*;
import org.springframework.statemachine.guard.Guard;
import org.springframework.statemachine.listener.StateMachineListener;
import org.springframework.statemachine.listener.StateMachineListenerAdapter;
import org.springframework.statemachine.persist.StateMachineRuntimePersister;
import org.springframework.statemachine.state.State;

/**
 * 订单状态机配置
 * 定义状态、事件、转换规则和监听器
 */
@Slf4j
@Configuration
@EnableStateMachineFactory
public class OrderStateMachineConfig extends StateMachineConfigurerAdapter<OrderState, OrderEvent> {

    private final StateMachineRuntimePersister<OrderState, OrderEvent, String> stateMachineRuntimePersister;

    public OrderStateMachineConfig(
            StateMachineRuntimePersister<OrderState, OrderEvent, String> stateMachineRuntimePersister) {
        this.stateMachineRuntimePersister = stateMachineRuntimePersister;
    }

    /**
     * 配置状态
     */
    @Override
    public void configure(StateMachineStateConfigurer<OrderState, OrderEvent> states) throws Exception {
        states
                .withStates()
                .initial(OrderState.UNPAID)
                .state(OrderState.WAITING_PAYMENT_CONFIRM)
                .state(OrderState.PAID, paidAction(), null) // 进入状态时执行动作
                .state(OrderState.PACKAGED)
                .state(OrderState.SHIPPED, shippedAction(), null)
                .state(OrderState.DELIVERED)
                .state(OrderState.CONFIRMED)
                .end(OrderState.COMPLETED)
                .end(OrderState.CANCELLED)
                .end(OrderState.REFUNDED)
                .end(OrderState.EXPIRED)
                .state(OrderState.REFUND_FAILED);
    }

    /**
     * 配置状态转换
     */
    @Override
    public void configure(StateMachineTransitionConfigurer<OrderState, OrderEvent> transitions) throws Exception {
        transitions
                // ========== 支付流程 ==========
                .withExternal()
                .source(OrderState.UNPAID)
                .target(OrderState.WAITING_PAYMENT_CONFIRM)
                .event(OrderEvent.PAY)
                .action(paymentInitiatedAction())
                .and()

                .withExternal()
                .source(OrderState.WAITING_PAYMENT_CONFIRM)
                .target(OrderState.PAID)
                .event(OrderEvent.PAY_SUCCESS)
                .guard(paymentSuccessGuard())
                .and()

                .withExternal()
                .source(OrderState.WAITING_PAYMENT_CONFIRM)
                .target(OrderState.UNPAID)
                .event(OrderEvent.PAY_FAILED)
                .action(paymentFailedAction())
                .and()

                .withExternal()
                .source(OrderState.UNPAID)
                .target(OrderState.CANCELLED)
                .event(OrderEvent.PAY_TIMEOUT)
                .action(timeoutAction())
                .and()

                .withExternal()
                .source(OrderState.UNPAID)
                .target(OrderState.EXPIRED)
                .event(OrderEvent.ORDER_EXPIRE)
                .and()

                // ========== 订单处理流程 ==========
                .withExternal()
                .source(OrderState.PAID)
                .target(OrderState.PACKAGED)
                .event(OrderEvent.PACKAGE)
                .guard(canPackageGuard())
                .action(packageAction())
                .and()

                .withExternal()
```

```java
package com.kidsky.service;

import com.kidsky.statemachine.domain.OrderEvent;
import com.kidsky.statemachine.domain.OrderState;
import com.kidsky.statemachine.entity.Order;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.messaging.Message;
import org.springframework.messaging.support.MessageBuilder;
import org.springframework.statemachine.StateMachine;
import org.springframework.statemachine.config.StateMachineFactory;
import org.springframework.statemachine.support.DefaultStateMachineContext;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDateTime;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

@Slf4j
@Service
@RequiredArgsConstructor
@Transactional
public class OrderStateMachineService {

    private final StateMachineFactory<OrderState, OrderEvent> stateMachineFactory;
    private final OrderService orderService;

    /**
     * 创建订单并初始化状态机
     */
    public Order createOrder(Order order) {
        // 1. 保存订单
        Order savedOrder = orderService.save(order);
        log.info("订单创建成功: ID={}, OrderNumber={}", savedOrder.getId(), savedOrder.getOrderNumber());

        // 2. 初始化状态机
        String machineId = getMachineId(savedOrder.getId());
        StateMachine<OrderState, OrderEvent> stateMachine = stateMachineFactory.getStateMachine(machineId);

        // 3. 启动状态机
        stateMachine.start();
        log.info("状态机初始化完成: machineId={}", machineId);

        return savedOrder;
    }

    /**
     * 发送事件到状态机
     */
    public boolean sendEvent(Long orderId, OrderEvent event) {
        return sendEvent(orderId, event, new HashMap<>());
    }

    /**
     * 发送事件到状态机（带参数）
     */
    public boolean sendEvent(Long orderId, OrderEvent event, Map<String, Object> headers) {
        try {
            // 1. 获取订单
            Order order = orderService.findById(orderId)
                    .orElseThrow(() -> new RuntimeException("订单不存在: " + orderId));

            // 2. 获取状态机实例
            String machineId = getMachineId(orderId);
            StateMachine<OrderState, OrderEvent> stateMachine = stateMachineFactory.getStateMachine(machineId);

            log.info("发送事件: orderId={}, event={}, currentState={}",
                    orderId, event, stateMachine.getState() != null ? stateMachine.getState().getId() : "None");

            // 3. 构建消息
            MessageBuilder<OrderEvent> messageBuilder = MessageBuilder.withPayload(event);

            // 添加标准头信息
            messageBuilder.setHeader("orderId", orderId);
            messageBuilder.setHeader("orderNumber", order.getOrderNumber());
            messageBuilder.setHeader("machineId", machineId);

            // 添加自定义头信息
            if (headers != null) {
                headers.forEach(messageBuilder::setHeader);
            }

            Message<OrderEvent> message = messageBuilder.build();

            // 4. 发送事件
            boolean accepted = stateMachine.sendEvent(message);

            if (accepted) {
                // 5. 状态转换成功，更新订单状态
                OrderState newState = stateMachine.getState().getId();
                order.setState(newState);

                // 根据状态更新相关时间
                updateOrderTimestamps(order, newState);

                orderService.save(order);
                log.info("事件处理成功: orderId={}, event={}, newState={}",
                        orderId, event, newState);
            } else {
                log.warn("事件被拒绝: orderId={}, event={}, currentState={}",
                        orderId, event, stateMachine.getState().getId());
            }

            return accepted;

        } catch (Exception e) {
            log.error("处理事件失败: orderId={}, event={}", orderId, event, e);
            throw new RuntimeException("状态机事件处理失败: " + e.getMessage(), e);
        }
    }

    /**
     * 获取订单当前状态
     */
    public OrderState getCurrentState(Long orderId) {
        String machineId = getMachineId(orderId);
        StateMachine<OrderState, OrderEvent> stateMachine = stateMachineFactory.getStateMachine(machineId);

        return stateMachine.getState() != null ? stateMachine.getState().getId() : null;
    }

    /**
     * 检查是否可以发送某个事件
     */
    public boolean canSendEvent(Long orderId, OrderEvent event) {
        OrderState currentState = getCurrentState(orderId);
        if (currentState == null) {
            return false;
        }

        String machineId = getMachineId(orderId);
        StateMachine<OrderState, OrderEvent> stateMachine = stateMachineFactory.getStateMachine(machineId);

        return stateMachine.getTransitions().stream()
                .anyMatch(transition ->
                        transition.getSource().getId() == currentState &&
                                transition.getTrigger() != null &&
                                transition.getTrigger().getEvent() == event);
    }

    /**
     * 重置状态机到指定状态（管理员功能，用于异常恢复）
     */
    public void resetStateMachine(Long orderId, OrderState targetState) {
        try {
            // 1. 获取订单
            Order order = orderService.findById(orderId)
                    .orElseThrow(() -> new RuntimeException("订单不存在: " + orderId));

            // 2. 重置状态机
            String machineId = getMachineId(orderId);
            StateMachine<OrderState, OrderEvent> stateMachine = stateMachineFactory.getStateMachine(machineId);

            stateMachine.stop();

            stateMachine.getStateMachineAccessor()
                    .doWithAllRegions(access ->
                            access.resetStateMachine(new DefaultStateMachineContext<>(targetState, null, null, null))
                    );

            stateMachine.start();

            // 3. 更新订单状态
            order.setState(targetState);
            orderService.save(order);

            log.info("状态机重置成功: orderId={}, targetState={}", orderId, targetState);

        } catch (Exception e) {
            log.error("重置状态机失败: orderId={}", orderId, e);
            throw new RuntimeException("状态机重置失败: " + e.getMessage(), e);
        }
    }

    /**
     * 获取状态机的完整历史（需要扩展实现）
     */
    public List<String> getStateHistory(Long orderId) {
        // 实际项目中可以从专门的审计表中查询
        log.info("获取订单 {} 的状态历史", orderId);
        return List.of("UNPAID -> PAID", "PAID -> SHIPPED");
    }

    /**
     * 生成状态机ID
     */
    private String getMachineId(Long orderId) {
        return "ORDER_" + orderId;
    }

    /**
     * 根据状态更新订单时间戳
     */
    private void updateOrderTimestamps(Order order, OrderState newState) {
        LocalDateTime now = LocalDateTime.now();

        switch (newState) {
            case PAID:
                order.setPayTime(now);
                break;
            case SHIPPED:
                order.setShipTime(now);
                break;
            case COMPLETED:
                order.setCompleteTime(now);
                break;
            default:
                break;
        }
    }
}
```

```java
    private static final int PAYMENT_TIMEOUT_MINUTES = 30; // 支付超时时间（分钟）

    /**
     * 每分钟检查一次支付超时的订单
     */
    @Scheduled(fixedDelay = 60000) // 1分钟
    @Transactional
    public void checkPaymentTimeout() {
        log.info("开始检查支付超时的订单...");
        
        LocalDateTime timeoutTime = LocalDateTime.now().minusMinutes(PAYMENT_TIMEOUT_MINUTES);
        List<Order> timeoutOrders = orderService.findTimeoutOrders(OrderState.UNPAID, PAYMENT_TIMEOUT_MINUTES);
        
        if (timeoutOrders.isEmpty()) {
            log.debug("未发现支付超时的订单");
            return;
        }
        
        log.info("发现 {} 个支付超时的订单", timeoutOrders.size());
        
        int successCount = 0;
        int failCount = 0;
        
        for (Order order : timeoutOrders) {
            try {
                log.info("处理支付超时订单: ID={}, OrderNumber={}, CreateTime={}", 
                    order.getId(), order.getOrderNumber(), order.getCreateTime());
                
                boolean accepted = stateMachineService.sendEvent(order.getId(), OrderEvent.PAY_TIMEOUT);
                
                if (accepted) {
                    successCount++;
                    log.info("订单支付超时处理成功: ID={}", order.getId());
                } else {
                    failCount++;
                    log.warn("订单支付超时处理失败（事件被拒绝）: ID={}", order.getId());
                }
                
            } catch (Exception e) {
                failCount++;
                log.error("处理支付超时订单异常: ID={}", order.getId(), e);
            }
        }
        
        log.info("支付超时处理完成: 成功={}, 失败={}, 总计={}", 
            successCount, failCount, timeoutOrders.size());
    }
    
    /**
     * 每天凌晨2点检查自动确认收货
     */
    @Scheduled(cron = "0 0 2 * * ?") // 每天凌晨2点
    @Transactional
    public void autoConfirmReceipt() {
        log.info("开始检查自动确认收货的订单...");
        
        // 查找所有已送达但未确认的订单
        List<Order> deliveredOrders = orderService.findByState(OrderState.DELIVERED);
        
        if (deliveredOrders.isEmpty()) {
            log.debug("未发现需要自动确认收货的订单");
            return;
        }
        
        LocalDateTime confirmTime = LocalDateTime.now().minusDays(AUTO_CONFIRM_DAYS);
        int successCount = 0;
        int failCount = 0;
        
        for (Order order : deliveredOrders) {
            // 检查是否送达超过指定天数
            if (order.getUpdateTime().isBefore(confirmTime)) {
                try {
                    log.info("处理自动确认收货订单: ID={}, OrderNumber={}, DeliverTime={}", 
                        order.getId(), order.getOrderNumber(), order.getUpdateTime());
                    
                    boolean accepted = stateMachineService.sendEvent(order.getId(), OrderEvent.AUTO_CONFIRM);
                    
                    if (accepted) {
                        successCount++;
                        log.info("订单自动确认收货成功: ID={}", order.getId());
                    } else {
                        failCount++;
                        log.warn("订单自动确认收货失败（事件被拒绝）: ID={}", order.getId());
                    }
                    
                } catch (Exception e) {
                    failCount++;
                    log.error("处理自动确认收货订单异常: ID={}", order.getId(), e);
                }
            }
        }
        
        log.info("自动确认收货处理完成: 成功={}, 失败={}, 检查总数={}", 
            successCount, failCount, deliveredOrders.size());
    }
```

### 6.修改yml

当第一次启动完成，数据库表已经创建好后，**第二次及以后启动时需要修改 `application.yml` 中的 `ddl-auto` 配置**，以防止意外修改表结构。当第一次启动完成，数据库表已经创建好后，**第二次及以后启动时需要修改 `application.yml` 中的 `ddl-auto` 配置**，以防止意外修改表结构。

~~~yml
spring:
  # ... 其他配置保持不变 ...
  jpa:
    hibernate:
      ddl-auto: validate  # ✅ 从 update 改为 validate
    show-sql: false       # ✅ 生产环境关闭SQL日志
~~~

## 关键特性说明

### 1. Spring State Machine 4.x 新特性

- **简化的持久化配置**：不再需要手动定义`StateMachineContextEntity`
- **自动状态恢复**：框架透明处理状态机的持久化和恢复
- **改进的API**：更简洁的配置方式和更好的类型安全

### 2. 事务管理

- 使用`@Transactional`确保状态转换和业务数据更新的一致性
- 状态机事件处理在事务中执行，失败时自动回滚

### 3. 错误处理

- 统一的API响应格式
- 详细的日志记录
- 异常情况的优雅降级

### 4. 可扩展性

- 守卫条件（Guard）允许添加业务规则检查
- 动作（Action）支持自定义业务逻辑
- 支持通过监听器监控状态机行为

## 常见问题解决

### 1. 启动时报`ClassNotFoundException`

确保依赖版本正确：

- Spring Boot 3.5.9
- Spring State Machine 4.0.0
- 所有`javax.persistence`已改为`jakarta.persistence`

### 2. 状态机事件被拒绝

检查：

- 当前状态是否支持该事件
- 守卫条件是否返回`true`
- 业务逻辑是否有异常

### 3. 状态未持久化

检查：

- 数据库连接配置是否正确
- `state_machine_context`表是否创建
- JPA配置是否正确

这个完整的实现方案基于最新的Spring Boot 3.5.9和Spring State Machine 4.0.0，可以直接用于生产环境或作为学习参考。
